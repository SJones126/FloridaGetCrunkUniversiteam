import java.awt.event.KeyEvent;

import java.awt.event.KeyListener;

public class Text extends Window implements KeyListener{
	
	public static boolean decimal = false;
	
Text(){
		
		textBox.setText("" + tempreture); // set this way because that is room tempture
	
	}

	@Override
	public void keyPressed(KeyEvent arg0) {}

	@Override
	public void keyReleased(KeyEvent e) {}

	@Override
	public void keyTyped(KeyEvent e){
		
		
		
char keyPressed = e.getKeyChar();
		
		
		if(keyPressed >= '0'  && keyPressed <= '9'){
			
			if(decimal == false){
			
			 temp += keyPressed;
			 
			 tempreture = Float.parseFloat(temp);
			 
			 slide.setValue((int) tempreture);
			 
			 textBox.setText("" + tempreture);
			 
			}else{
				
				temp += "." + keyPressed;
				
				tempreture = Float.parseFloat(temp);
				
				textBox.setText("" + tempreture);
				
				decimal = false;
				
			}
			 
		}
		if(keyPressed == '.'){
			
			decimal = true;
			
		}
			
		 else{
				e.consume();
		 }
	}
	
}
